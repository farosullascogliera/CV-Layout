\newcommand{\entry}[4]{%
    {\large\textbf{#1}} \hfill #3 \\%

% Job entry command with colored title, subtitle, right-aligned location and date
\newcommand{\jobentry}[4]{%
    {\Large\textbf{#1}} \hfill {\large\textbf{#3}} \\%
    \rlap{\large\textbf{#2}} \hfill #4%
}

% Smaller entry command with colored title, subtitle, right-aligned location and date
\newcommand{\entrySmall}[4]{
    {\scshape\large\textbf{#1}} \hfill #3 \\
    #2 \hfill #4
}

\newcommand{\tableentry}[3]{
    \textsc{#1} & #2
    \ifstrequal{#3}{}{\\}{\\[6pt]}
}

\newcommand{\jobpoint}[2]{
